from datetime import datetime, timezone, timedelta
from faker import Faker
import json, random, uuid
fake = Faker()
TENANT = "curry_bowl_express"
LOCATIONS = ["mission", "castro", "soma", "oakland", "berkeley", "sunnyvale", "fremont"]
SOURCES = ["google", "reddit", "doordash", "ubereats", "survey", "qr", "voice"]
SCENARIOS = ["cold food", "missing item", "late delivery", "rude staff", "great biryani", "portion size", "price concern"]
def make_event(i: int) -> dict:
    scenario = random.choice(SCENARIOS)
    return {"event_id": str(uuid.uuid4()), "tenant_id": TENANT, "location_id": random.choice(LOCATIONS), "source": random.choice(SOURCES), "source_event_id": f"src-{i}", "review_text": f"{fake.sentence()} Main topic: {scenario}.", "rating": round(random.uniform(1, 5), 1), "event_ts": (datetime.now(timezone.utc) - timedelta(minutes=random.randint(1, 50000))).isoformat(), "raw_payload": {"scenario": scenario}}
if __name__ == "__main__":
    with open("data/dummy/out/feedback_events.jsonl", "w") as f:
        for i in range(25000):
            f.write(json.dumps(make_event(i)) + "\n")
    print("✅ wrote data/dummy/out/feedback_events.jsonl")
python — Kafka producerservices/ingestion-worker/producer.py
import json, os
from confluent_kafka import Producer
from shared.schemas.feedback_event import FeedbackEvent
producer = Producer({"bootstrap.servers": os.getenv("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092"), "enable.idempotence": True})
def delivery_report(err, msg):
    print(f"delivery={'failed' if err else 'ok'} key={msg.key().decode() if msg and msg.key() else ''}")
with open("data/dummy/out/feedback_events.jsonl") as f:
    for line in f:
        event = FeedbackEvent(**json.loads(line))
        producer.produce("feedback.raw.events", key=event.event_id, value=event.model_dump_json(), callback=delivery_report)
        producer.poll(0)
producer.flush()
python — Ingestion workerservices/ingestion-worker/consumer.py
import os, json
from confluent_kafka import Consumer
consumer = Consumer({"bootstrap.servers": os.getenv("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092"), "group.id": "fip-ingestion-worker", "auto.offset.reset": "earliest", "enable.auto.commit": False})
consumer.subscribe(["feedback.raw.events"])
while True:
    msg = consumer.poll(1.0)
    if msg is None: continue
    if msg.error(): print(msg.error()); continue
    event = json.loads(msg.value())
    print(f"bronze-write-ready event_id={event['event_id']} source={event['source']}")
    consumer.commit(msg)
