Overview

What this phase does and why it exists

Consumes

None. This is the first phase.

Adds

Creates the repo, Terraform, Helm, shared schema, logging, topics, secrets, and local AWS-parity foundation.

A running FLOCI-backed foundation that Phase 1 reuses.

makefile — Create MakefileMakefile
SHELL := /bin/bash
.ONESHELL:
.DEFAULT_GOAL := help

AWS_REGION ?= us-west-2
FLOCI_ENDPOINT ?= http://localhost:4566
TF_DIR := infra/terraform/envs/local-floci
HELM_CHART := infra/helm/feedback-platform
NAMESPACE := feedback
CLUSTER_NAME := feedback-eks-local

.PHONY: help doctor bootstrap floci-start tf-check tf-init tf-plan tf-apply secrets-inject k8s-config topics helm-install verify clean tf-destroy

help:
	@echo "Phase 0 commands:"
	@echo "  make doctor      # verify tools"
	@echo "  make bootstrap   # start FLOCI -> terraform -> secrets -> topics -> helm -> verify"
	@echo "  make verify      # re-run verification"
	@echo "  make clean       # stop FLOCI"

doctor:
	@command -v floci >/dev/null || (echo "❌ floci not installed"; exit 1)
	@command -v aws >/dev/null || (echo "❌ aws cli not installed"; exit 1)
	@command -v terraform >/dev/null || (echo "❌ terraform not installed"; exit 1)
	@command -v kubectl >/dev/null || echo "⚠️ kubectl missing; Kubernetes verification may be skipped"
	@command -v helm >/dev/null || echo "⚠️ helm missing; Helm install may be skipped"
	@echo "✅ Local tool check complete"

bootstrap: doctor floci-start tf-check tf-init tf-apply secrets-inject k8s-config topics helm-install verify
	@echo "✅ Phase 0 foundation is running"

floci-start:
	floci start
	aws --endpoint-url $(FLOCI_ENDPOINT) sts get-caller-identity
	aws --endpoint-url $(FLOCI_ENDPOINT) s3 ls || true

tf-check:
	@test -f $(TF_DIR)/versions.tf || (echo "❌ Missing $(TF_DIR)/versions.tf. Create all Terraform files from the guide first."; exit 1)
	@test -f $(TF_DIR)/provider.tf || (echo "❌ Missing $(TF_DIR)/provider.tf. Create all Terraform files from the guide first."; exit 1)
	@test -f $(TF_DIR)/main.tf || (echo "❌ Missing $(TF_DIR)/main.tf. Create all Terraform files from the guide first."; exit 1)
	@echo "✅ Terraform config files found"

tf-init:
	cd $(TF_DIR)
	terraform init -upgrade

tf-plan:
	cd $(TF_DIR)
	terraform plan

tf-apply:
	cd $(TF_DIR)
	terraform apply -auto-approve
	terraform output || true

secrets-inject:
	aws --endpoint-url $(FLOCI_ENDPOINT) secretsmanager put-secret-value --secret-id /feedback/local/openai_api_key --secret-string "sk-dummy-local-key-replace-in-prod" || true
	aws --endpoint-url $(FLOCI_ENDPOINT) secretsmanager put-secret-value --secret-id /feedback/local/jwt_secret --secret-string "local-jwt-secret-$$(openssl rand -hex 16)" || true
	aws --endpoint-url $(FLOCI_ENDPOINT) secretsmanager put-secret-value --secret-id /feedback/local/db_password --secret-string "local-db-password-$$(openssl rand -hex 8)" || true

k8s-config:
	aws --endpoint-url $(FLOCI_ENDPOINT) eks update-kubeconfig --region $(AWS_REGION) --name $(CLUSTER_NAME) || true
	kubectl get nodes || true

topics:
	chmod +x scripts/create_topics.sh
	./scripts/create_topics.sh || true

helm-install:
	kubectl create namespace $(NAMESPACE) --dry-run=client -o yaml | kubectl apply -f - || true
	helm upgrade --install feedback-platform $(HELM_CHART) \
	  --namespace $(NAMESPACE) --create-namespace \
	  -f $(HELM_CHART)/values-local.yaml \
	  --wait --timeout 120s || true

verify:
	@echo "🔎 S3 buckets"
	aws --endpoint-url $(FLOCI_ENDPOINT) s3 ls || true
	@echo "🔎 Secrets"
	aws --endpoint-url $(FLOCI_ENDPOINT) secretsmanager list-secrets --query 'SecretList[].Name' || true
	@echo "🔎 Kubernetes"
	kubectl get pods -n $(NAMESPACE) || true
	@echo "🔎 Kafka topics"
	kafka-topics.sh --list --bootstrap-server localhost:9092 || true

tf-destroy:
	cd $(TF_DIR)
	terraform destroy -auto-approve

clean:
	floci stop

bash — Create Kafka topic scriptscripts/create_topics.sh
#!/usr/bin/env bash
set -euo pipefail
BROKER="${KAFKA_BOOTSTRAP_SERVERS:-localhost:9092}"
TOPICS=("feedback.raw.events" "feedback.bronze.events" "feedback.silver.events" "feedback.gold.metrics" "feedback.agent.actions" "feedback.dlq.events")

if ! command -v kafka-topics.sh >/dev/null 2>&1; then
  echo "⚠️ kafka-topics.sh not found. Install Kafka CLI or use FLOCI/MSK topic tooling."
  exit 0
fi

for topic in "${TOPICS[@]}"; do
  kafka-topics.sh --bootstrap-server "$BROKER" --create --if-not-exists --topic "$topic" --partitions 3 --replication-factor 1 --config retention.ms=604800000
done

kafka-topics.sh --bootstrap-server "$BROKER" --list
python — Create shared schemashared/schemas/feedback_event.py
from datetime import datetime, timezone
from typing import Literal, Optional
from pydantic import BaseModel, Field, field_validator
import uuid

Source = Literal["google", "reddit", "doordash", "ubereats", "survey", "qr", "voice"]

class FeedbackEvent(BaseModel):
    event_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    tenant_id: str
    location_id: str
    source: Source
    source_event_id: str
    review_text: str
    rating: Optional[float] = None
    event_ts: datetime
    ingested_at: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))
    raw_payload: dict = Field(default_factory=dict)

    @field_validator("review_text")
    @classmethod
    def text_must_not_be_empty(cls, value: str) -> str:
        value = value.strip()
        if len(value) < 3:
            raise ValueError("review_text must contain at least 3 characters")
        return value[:10000]
python — Create shared loggershared/logging/json_logger.py
import logging, json, sys
from datetime import datetime, timezone

class JsonFormatter(logging.Formatter):
    def format(self, record):
        return json.dumps({"ts": datetime.now(timezone.utc).isoformat(), "level": record.levelname, "logger": record.name, "message": record.getMessage()})

def get_logger(name: str):
    logger = logging.getLogger(name)
    if logger.handlers:
        return logger
    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(JsonFormatter())
    logger.addHandler(handler)
    logger.setLevel(logging.INFO)
    return logger
