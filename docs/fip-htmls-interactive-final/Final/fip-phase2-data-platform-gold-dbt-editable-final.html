python — data quality rulesservices/data-platform-worker/dq_rules.py
from datetime import datetime

REQUIRED_FIELDS = ["event_id", "tenant_id", "location_id", "source", "source_event_id", "review_text", "event_ts"]

def validate_event(payload):
    errors = []
    for field in REQUIRED_FIELDS:
        if field not in payload or payload[field] in (None, ""):
            errors.append(f"missing_required_field:{field}")
    try:
        r = payload.get("rating")
        if r is not None and not (1 <= float(r) <= 5):
            errors.append("rating_out_of_range")
    except Exception:
        errors.append("rating_not_numeric")
    try:
        datetime.fromisoformat(str(payload.get("event_ts")).replace("Z", "+00:00"))
    except Exception:
        errors.append("invalid_event_ts")
    if len(str(payload.get("review_text", "")).strip()) < 3:
        errors.append("review_text_too_short")
    return len(errors) == 0, errors
python — cleanerservices/data-platform-worker/cleaner.py
import re
EMAIL_RE = re.compile(r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}\b")
PHONE_RE = re.compile(r"\b(?:\+?1[-.\s]?)?\(?\d{3}\)?[-.\s]?\d{3}[-.\s]?\d{4}\b")

def normalize_text(text):
    text = EMAIL_RE.sub("[EMAIL_MASKED]", text or "")
    text = PHONE_RE.sub("[PHONE_MASKED]", text)
    return re.sub(r"\s+", " ", text).strip()

def clean_event(payload, quality_errors):
    return {**payload, "clean_review_text": normalize_text(str(payload.get("review_text", ""))), "quality_errors": quality_errors, "processing_stage": "silver"}
python — NLP enricherservices/data-platform-worker/nlp_enricher.py
import re
CATEGORY_RULES = {"delivery":["late","delivery","driver","cold"],"order_accuracy":["missing","wrong","item","order"],"service":["rude","staff","service","wait"],"food_quality":["fresh","taste","spicy","biryani","portion"],"pricing":["price","expensive","cost"]}

def keywords(text):
    words = re.findall(r"[a-zA-Z]{4,}", text.lower())
    stop = {"this","that","with","main","topic","location","food","very"}
    out=[]
    for w in words:
        if w not in stop and w not in out: out.append(w)
        if len(out)>=8: break
    return out

def categorize(text):
    t = text.lower(); scores = {c:sum(1 for term in terms if term in t) for c,terms in CATEGORY_RULES.items()}
    best=max(scores,key=scores.get); return best if scores[best]>0 else "general"

def sentiment_from_rating(rating):
    try:
        r=float(rating)
        return "negative" if r<=2 else "neutral" if r<=3.5 else "positive"
    except Exception: return "unknown"

def enrich_nlp(silver):
    text=silver.get("clean_review_text") or silver.get("review_text","")
    category=categorize(text); sentiment=sentiment_from_rating(silver.get("rating"))
    severity="high" if sentiment=="negative" and category in {"delivery","order_accuracy","service"} else "medium" if sentiment=="negative" else "low"
    return {**silver,"category":category,"sentiment":sentiment,"severity":severity,"intent":"complaint" if sentiment=="negative" else "praise" if sentiment=="positive" else "feedback","keywords":keywords(text),"embedding_text":f"{category} | {sentiment} | {text}","processing_stage":"gold"}
python — main workerservices/data-platform-worker/worker.py
import json, os, csv
from pathlib import Path
import boto3
from services.data_platform_worker.cleaner import clean_event
from services.data_platform_worker.dq_rules import validate_event
from services.data_platform_worker.nlp_enricher import enrich_nlp

AWS_ENDPOINT_URL=os.getenv("AWS_ENDPOINT_URL","http://localhost:4566"); AWS_REGION=os.getenv("AWS_REGION","us-west-2")
RAW_BUCKET=os.getenv("RAW_BUCKET","fip-local-raw"); PROCESSED_BUCKET=os.getenv("PROCESSED_BUCKET","fip-local-processed")
LOCAL_INPUT=os.getenv("LOCAL_INPUT","data/dummy/out/feedback_events.jsonl"); MAX_RECORDS=int(os.getenv("MAX_RECORDS","25000")); WORK_DIR=Path(os.getenv("WORK_DIR","/app/work"))

def s3_client(): return boto3.client("s3",endpoint_url=AWS_ENDPOINT_URL,aws_access_key_id=os.getenv("AWS_ACCESS_KEY_ID","test"),aws_secret_access_key=os.getenv("AWS_SECRET_ACCESS_KEY","test"),region_name=AWS_REGION)
def write_s3(key,payload): s3_client().put_object(Bucket=PROCESSED_BUCKET,Key=key,Body=json.dumps(payload,default=str).encode("utf-8"))
def read_local_jsonl(path):
    p=Path(path)
    if p.exists():
        for line in p.open(encoding="utf-8"):
            if line.strip(): yield json.loads(line)
def read_raw_s3():
    c=s3_client(); resp=c.list_objects_v2(Bucket=RAW_BUCKET)
    for obj in resp.get("Contents",[]):
        if obj["Key"].endswith(".json"): yield json.loads(c.get_object(Bucket=RAW_BUCKET,Key=obj["Key"])["Body"].read())
def write_gold_csv(path, rows):
    path.parent.mkdir(parents=True,exist_ok=True); fields=["event_id","tenant_id","location_id","source","rating","event_ts","sentiment","category","severity","intent","clean_review_text","embedding_text"]
    with path.open("w",newline="",encoding="utf-8") as f:
        w=csv.DictWriter(f,fieldnames=fields); w.writeheader(); [w.writerow({k:r.get(k,"") for k in fields}) for r in rows]
def main():
    events=list(read_local_jsonl(LOCAL_INPUT)) or list(read_raw_s3()); silver=[]; gold=[]; invalid=[]
    for event in events[:MAX_RECORDS]:
        ok,errors=validate_event(event)
        if not ok: invalid.append({**event,"quality_errors":errors}); continue
        s=clean_event(event,errors); g=enrich_nlp(s); silver.append(s); gold.append(g)
        tenant=g.get("tenant_id","unknown"); source=g.get("source","unknown"); eid=g.get("event_id")
        write_s3(f"silver/tenant={tenant}/source={source}/event_id={eid}.json",s); write_s3(f"gold/tenant={tenant}/source={source}/event_id={eid}.json",g)
    for q in invalid: write_s3(f"quarantine/tenant={q.get('tenant_id','unknown')}/event_id={q.get('event_id','missing')}.json",q)
    csv_path=WORK_DIR/"gold_feedback.csv"; write_gold_csv(csv_path,gold)
    s3_client().put_object(Bucket=PROCESSED_BUCKET,Key="dbt_input/gold_feedback.csv",Body=csv_path.read_bytes())
    print(f"✅ phase2_worker_complete silver={len(silver)} gold={len(gold)} quarantine={len(invalid)}")
if __name__=="__main__": main()

import os, subprocess
from pathlib import Path
import boto3
AWS_ENDPOINT_URL=os.getenv("AWS_ENDPOINT_URL","http://localhost:4566"); PROCESSED_BUCKET=os.getenv("PROCESSED_BUCKET","fip-local-processed")
WORK_DIR=Path(os.getenv("WORK_DIR","/app/work")); PROJECT_DIR=Path("data/dbt/feedback_analytics")
def s3_client(): return boto3.client("s3",endpoint_url=AWS_ENDPOINT_URL,aws_access_key_id=os.getenv("AWS_ACCESS_KEY_ID","test"),aws_secret_access_key=os.getenv("AWS_SECRET_ACCESS_KEY","test"),region_name=os.getenv("AWS_REGION","us-west-2"))
def main():
    WORK_DIR.mkdir(parents=True,exist_ok=True)
    obj=s3_client().get_object(Bucket=PROCESSED_BUCKET,Key="dbt_input/gold_feedback.csv")
    (WORK_DIR/"gold_feedback.csv").write_bytes(obj["Body"].read())
    subprocess.check_call(["dbt","run","--project-dir",str(PROJECT_DIR),"--profiles-dir",str(PROJECT_DIR/"profiles")])
    mart=PROJECT_DIR/"target"/"feedback_analytics.duckdb"
    if mart.exists(): s3_client().put_object(Bucket=PROCESSED_BUCKET,Key="marts/feedback_analytics.duckdb",Body=mart.read_bytes())
    print("✅ dbt_marts_complete")
if __name__=="__main__": main()

-- data/dbt/feedback_analytics/models/staging/stg_gold_feedback.sql
with src as (select * from read_csv_auto('/app/work/gold_feedback.csv', header=true))
select event_id, tenant_id, location_id, source, cast(rating as double) as rating, event_ts, sentiment, category, severity, intent, clean_review_text, embedding_text from src;

-- data/dbt/feedback_analytics/models/gold/fact_feedback.sql
select *, case when sentiment = 'negative' then 1 else 0 end as is_complaint from {{ ref('stg_gold_feedback') }};

-- data/dbt/feedback_analytics/models/marts/mart_location_feedback_summary.sql
select tenant_id, location_id, category, sentiment, count(*) as feedback_count, avg(rating) as avg_rating, sum(is_complaint) as complaint_count from {{ ref('fact_feedback') }} group by 1,2,3,4;

-- data/dbt/feedback_analytics/models/marts/mart_complaint_trends.sql
select tenant_id, location_id, category, date_trunc('day', cast(event_ts as timestamp)) as feedback_day, count(*) as feedback_count, sum(is_complaint) as complaint_count from {{ ref('fact_feedback') }} group by 1,2,3,4;

-- data/dbt/feedback_analytics/models/marts/mart_embedding_ready_feedback.sql
select event_id, tenant_id, location_id, source, category, sentiment, severity, intent, embedding_text from {{ ref('fact_feedback') }} where embedding_text is not null and length(embedding_text) > 10;

# -------------------------------------------------------------------
# Phase 2 — Silver → NLP Gold → dbt Analytics Mart on FLOCI EKS/S3

PHASE2_IMAGE ?= fip-local/data-platform-worker:phase2
PHASE2_IMAGE_REPO ?= fip-local/data-platform-worker
PHASE2_IMAGE_TAG ?= phase2
NAMESPACE ?= feedback
HELM_CHART ?= infra/helm/feedback-platform
.PHONY: phase2-install phase2-build phase2-deploy phase2-status phase2-logs phase2-verify phase2-clean phase2
phase2-install:
	python3 -m pip install -r services/data-platform-worker/requirements.txt
phase2-build:
	docker build -t $(PHASE2_IMAGE) -f services/data-platform-worker/Dockerfile .
phase2-deploy:
	kubectl delete job data-platform-worker dbt-analytics-mart -n $(NAMESPACE) --ignore-not-found=true
	kubectl create namespace $(NAMESPACE) --dry-run=client -o yaml | kubectl apply -f -
	helm upgrade --install feedback-platform ./$(HELM_CHART) \
	  --namespace $(NAMESPACE) \
	  -f $(HELM_CHART)/values-local.yaml \
	  --set dataPlatform.enabled=true \
	  --set dataPlatform.image.repository=$(PHASE2_IMAGE_REPO) \
	  --set dataPlatform.image.tag=$(PHASE2_IMAGE_TAG) \
	  --set dataPlatform.aws.endpointUrl=http://localhost:4566 \
	  --set dataPlatform.s3.rawBucket=fip-local-raw \
	  --set dataPlatform.s3.processedBucket=fip-local-processed \
	  --wait --timeout 240s
phase2-status:
	kubectl get pods -n $(NAMESPACE)
	kubectl get jobs -n $(NAMESPACE)
phase2-logs:
	kubectl logs -n $(NAMESPACE) job/data-platform-worker --tail=120 || true
	kubectl logs -n $(NAMESPACE) job/dbt-analytics-mart --tail=120 || true
phase2-verify:
	@echo "🔎 Silver objects"; aws --endpoint-url http://localhost:4566 s3 ls s3://fip-local-processed/silver --recursive | head || true
	@echo "🔎 Gold objects"; aws --endpoint-url http://localhost:4566 s3 ls s3://fip-local-processed/gold --recursive | head || true

	@echo "✅ Phase 2 verified: Silver + Gold + dbt mart ready for Phase 3 embeddings"
phase2-clean:
	kubectl delete job data-platform-worker dbt-analytics-mart -n $(NAMESPACE) --ignore-not-found=true
phase2: phase2-install phase2-build phase2-deploy phase2-status phase2-logs phase2-verify
